Understands type coercion and strategies to avoid it.

console.log("5" + 3); // "53" (String, ikke 8!)
console.log("5" - 3); // 2 (Number)
console.log(Number("5") + 3); // 8

console.log(5 == "5"); // true (pga. coercion)
console.log(5 === "5"); // false (ingen coercion)
